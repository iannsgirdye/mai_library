\subsection{Определение | Форма от $x$}
$P(x)$ ("форма от $x$") --- \{И, Л\} \\
$A =$ \{$x$ | $P(x) =$ И\} = \{$x$ | $P(x)$\} ("множество $A$ --- это множество таких $x$, что форма от $x$ истинна")

\subsubsection{Пример}
\begin{gather*}
    x^2 - 1 = 0 \\
    A = \{1, -1\} = \{x~|~x^2 - 1 = 0\}
\end{gather*}


\subsection{Свойства}
\begin{enumerate}
    \item $\{1, 2\} = \{2, 1, 1, 2, 1\}$ \\
    \item $A \subseteq B \Rightarrow \forall x \in A$ \\
    \item $A \subseteq B, B \subseteq C \Rightarrow A \subseteq C$ \\
    \item $A = B \Leftrightarrow A \subseteq B, B \subseteq A$
\end{enumerate}

\subsubsection{Пример}
\[
    \mathbb{N} \subseteq \mathbb{Z} \subseteq \mathbb{Q} \subseteq \mathbb{R} \subseteq \mathbb{C}
\]

\subsubsection{Пример}
\begin{gather*}
    A = \{1, 2\}, B = \{ \{1, 2\}, 3 \}, C = \{1, 2, 3\} \\
    1 \in A, 1 \notin B, A \subseteq C, B \neq C, A \in B
\end{gather*}

\section{Свойства операций (основные тождества)}

\subsection{Коммутативность (переместительный закон)}
\begin{gather*}
    A \cup B = B \cup A \\
    A \cap B = B \cap A \\
    A \oplus B = B \oplus A
\end{gather*}


\subsection{Ассоциативность (сочетательный закон)}
\begin{gather*}
    A \cup (B \cup C) = (A \cup B) \cup C = A \cup B \cup C \\
    A \cap (B \cap C) = (A \cap B) \cap C = A \cap B \cap C \\
    A \setminus (B \setminus C) = (A \setminus B) \setminus C
\end{gather*}


\subsection{Дистрибутивность (распределительный закон)}
\begin{gather*}
    A \cup (B \cap C) = (A \cup B) \cap (A \cup C) \\
    A \cap (B \cup C) = (A \cap B) \cup (A \cap C)
\end{gather*}

\subsubsection{Примечание}
Приоритет операций определяется только скобками.


\subsection{Закон идемпотентности}
\begin{gather*}
    A \cup A = A \\
    A \cap A = A
\end{gather*}


\subsection{Закон де Моргана}
\begin{gather*}
    \overline{A \cup B} = \overline{A} \cap \overline{B} \\
    \overline{A \cap B} = \overline{A} \cup \overline{B}
\end{gather*}


\subsection{Закон поглощения}
\begin{gather*}
    A \cup (A \cap B) = A \\
    A \cap (A \cup B) = A
\end{gather*}


\subsection{Расщепления}
\begin{gather*}
    A = (A \cup B) \cap (A \cup \overline{B})
\end{gather*}


\subsection{Двойное отрицание}
\begin{gather*}
    \overline{\overline{A}} = A \\
    X \setminus Y = X \cap \overline{Y} \\
    X \oplus Y = (X \cap \overline{Y}) \cup (\overline{X} \cap Y) = (X \cup Y) \cap (\overline{X} \cup \overline{Y})
\end{gather*}


\subsection{Разности}
\begin{gather*}
    A \setminus (B \cup C) = (A \setminus B) \cap (A \setminus C) \\
    A \setminus (B \cap C) = (A \setminus B) \cup (A \setminus C) \\ 
    (A \setminus B) \cap C = (A \cap C) \setminus (B \cap C) \\
    (A \setminus B) \setminus C = (A \setminus C) \setminus B
\end{gather*}



\section{Характеристическая функция множества}

\subsection{Определение}
\begin{gather*}
    A \subseteq \mathbb{U} \\
    A \Rightarrow \psi_A: \mathbb{U} \Rightarrow \{0, 1\} \\
    \psi_A(x) = 
    \begin{cases}
        1, \text{ если } x \in A \\
        0, \text{ если } x \notin A
    \end{cases}
\end{gather*}


\subsection{Функции основных операций через характеристические функции}
$\psi_A(x), \psi_B(x)$
\begin{enumerate}
    \item $\psi_{A \cap B} = \psi_A \cdot \psi_B$
    \item $\psi_{A \cup B} = \psi_A + \psi_B - \psi_A \cdot \psi_B$
    \item $\psi_{\overline{A}} = 1 - \psi_A$
    \item $\psi_{A \setminus B} = \psi_{A \cap \overline{B}} = \psi_A \cdot (1 - \psi_B) = \psi_A - \psi_A \psi_B$
    \item $\psi^2_A = \psi_A$
    \item $\psi_{A \oplus B} = \psi_A + \psi_B - 2 \psi_A \psi_B = \psi^2_A + \psi^2_B - 2 \psi_A \psi_B = (\psi_A - \psi_B)^2 = |\psi_A - \psi_B|$
\end{enumerate}

\subsubsection{Пример}
Доказать, что
\[
    A \setminus (B \setminus C) = (A \setminus B) \cup (A \cap C)
\]

\paragraph{Доказательство}
\begin{gather*}
    \psi_{A \setminus (B \setminus C)} = 
    \psi_A \cdot (1 - \psi_{B \setminus C}) = 
    \psi_A (1 - \psi_B (1 - \psi_C)) =
    \psi_A - \psi_A \psi_B + \psi_A \psi_B \psi_C \\
    \psi_{(A \setminus B) \cup (A \cap C)} = 
    \psi_A (1 - \psi_B) + \psi_A \psi_C - \psi_A (1 - \psi_B) \cdot \psi_A \psi_C = \\
    = \psi_A - \psi_A \psi_B + \psi_A \psi_C - \psi_A \psi_C + \psi_A \psi_B \psi_C = \psi_A - \psi_A \psi_B + \psi_A \psi_B \psi_C
\end{gather*}


\subsubsection{Пример}
Доказать, что
\[
    A \cap (B \oplus C) = (A \cap B) \oplus (A \cap C)
\]

\paragraph{Доказательство}
\begin{gather*}
    \psi_{A \cap (B \oplus C)} = \psi_A \psi_{B \oplus C} = \psi_A |\psi_B - \psi_C| \\
    \psi_{(A \cap B) \oplus (A \cap C)} = |\psi_{A \cap B} - \psi_{A \cap C}| = \psi_A |\psi_B - \psi_C|
\end{gather*}

\section{Методы утверждений}
\begin{enumerate}
    \item Теорема --- важный результат
    \item Утверждение --- теорема с менее важным результатом
    \item Лемма --- вспомогательное утверждение для теоремы
\end{enumerate}


\subsection{Утверждение}
3 эквивалентных условия (выполняются все при выполнении одного)
\begin{enumerate}
    \item $A \subseteq B$
    \item $A \cap B = A$
    \item $A \cup B = B$
\end{enumerate}

\subsubsection{Доказательство 1 $\Rightarrow$ 2}
\paragraph{Дано}
\[
    A \subseteq B
\]

\paragraph{Доказать}
\[
    A \cap B = A
\]

\paragraph{Доказательство}
\begin{gather*}
    x = y \Leftrightarrow x \subseteq y, y \subseteq x \\
    x \subseteq y: \forall x \in X \Rightarrow x \in y
\end{gather*}
\begin{enumerate}
    \item $A \cap B \subseteq A$
    \[
        \text{Пусть } x \in A \cap B \Rightarrow x \in A.
    \]
    \item $A \subseteq A \cap B$
    \[
        \text{Пусть } x \in A \underset{\text{усл.}}{\Rightarrow} x \in B \Rightarrow x \in A \cap B.
    \]
\end{enumerate}


\subsubsection{Доказательство 2 $\Rightarrow$ 3}
\paragraph{Дано}
\[
    A \cap B = A
\]

\paragraph{Доказать}
\[
    A \cup B = B
\]

\paragraph{Доказательство}
\[
    A \cup B = (A \cap B) \cup B = B \text{ --- поглощение}
\]


\subsubsection{Доказательство 3 $\Rightarrow$ 1}
\paragraph{Дано}
\[
    A \cup B = B
\]

\paragraph{Доказать}
\[
    A \subseteq B
\]

\paragraph{Доказательство (от противного)}
\begin{gather*}
    \text{Предположим, что } A \nsubseteq B \Rightarrow \exists x \in A \text{ и } x \notin B. \\ 
    \text{Так как } x \in A \Rightarrow x \in A \cup B \underset{\text{усл.}}{\Rightarrow} x \in B \text{ --- противоречие} \Rightarrow A \subseteq B
\end{gather*}

\section{Метод рассуждений}

\subsection{Пример | И. А. Лавров, Л. Л. Максимова: номер 13(г)}
Доказать, что 
\[
    A \subseteq B \cup C \Leftrightarrow A \cap \overline{B} \subseteq C
\]

\subsubsection{Доказательство 1 $\Rightarrow$ 2}
\paragraph{Дано}
\[
    A \subseteq B \cup C
\]

\paragraph{Доказать}
\[
    A \cap \overline{B} \subseteq C
\]

\paragraph{Доказательство}
\begin{gather*}
    \text{Пусть } x \in A \cap \overline{B} \Rightarrow x \in A \text{ и } x \in \overline B. \\
    x \in A \underset{\text{усл.}}{\Rightarrow} x \in B \cup C \Rightarrow x \in B \text{ или } x \in C, \text{ но } x \notin B \Rightarrow x \in C.
\end{gather*}


\subsubsection{Доказательство 1 $\Leftarrow$ 2}
\paragraph{Дано}
\[
    A \cap \overline{B} \subseteq C
\]

\paragraph{Доказать}
\[
    A \subseteq B \cup C
\]

\paragraph{Доказательство}
\begin{gather*}
    \text{Пусть } x \in A. \\
    \text{Если } x \in B \Rightarrow x \in B \cup C. \\
    \text{Если } x \notin B \Rightarrow x \in A \cap \overline B \underset{\text{усл.}}{\Rightarrow} x \in C \Rightarrow x \in B \cup C.
\end{gather*}


\subsection{Пример}
\begin{gather*}
    X = Y \Rightarrow X \cup Z = Y \cup Z \\
    X \cup Z = Y \cup Z \nRightarrow X = Y
\end{gather*}


\subsection{Пример}
Доказать, что
\[
    A \oplus B = B \setminus C \Leftrightarrow A = B \cap C
\]

\subsubsection{Доказательство 1 $\Leftarrow$ 2}
\paragraph{Дано}
\[
    A = B \cap C
\]

\paragraph{Доказать}
\[
    A \oplus B = B \setminus C
\]

\paragraph{Доказательство}
\begin{gather*}
    (B \cap C) \oplus B = (B \cap C \cap \overline{B}) \cup \left( (\overline{B \cap C}) \cap B \right) = 
    (\overline{B} \cup \overline{C}) \cap B = (\overline{B} \cap B) \cup (B \cap C) = B \cap C
\end{gather*}


\subsubsection{Доказательство 1 $\Rightarrow$ 2}
\paragraph{Дано}
\[
    A \oplus B = B \setminus C
\]

\paragraph{Доказать}
\[
    A = B \cap C
\]

\paragraph{Доказательство 1}
\begin{enumerate}
    \item Докажем, что $A \subseteq B \cap C$
    \begin{gather*}
        \text{Пусть } x \in A. \\
        \text{Если } x \in B \Rightarrow x \notin A \oplus B \underset{\text{усл.}}{\Rightarrow} 
        x \notin B \setminus C \Rightarrow x \in \overline{B \setminus C} = \overline{B \cap \overline{C}} = \overline{B} \cup C \Rightarrow \\
        x \in \overline{B} \text{ или } x \in C, \text{ но } x \in B \text{ --- противоречие } \Rightarrow x \in B.
    \end{gather*}
    \item Докажем, что $B \cap C \subseteq A$
    \begin{gather*}
    \text{Пусть } x \in B \cap C \Rightarrow x \in B \text{ и } x \in C \Rightarrow 
    x \notin B \setminus C \underset{\text{усл.}}{\Rightarrow} x \notin A \oplus B \Rightarrow x \in A.
    \end{gather*}
\end{enumerate}

\paragraph{Доказательство 2}
\begin{gather*}
    (A \oplus B) \oplus B = (B \setminus C) \oplus B \\
    A = (B \setminus C) \oplus B \\
    (B \setminus C) \oplus B = 
    (B \cap \overline{C} \cap \overline{B}) \cup \left( (\overline{B \cap \overline{C}}) \cap B \right) = 
    (\overline{B} \cup C) \cap B = 
    (B \cap \overline{B}) \cup (B \cap C) = B \cap C.
\end{gather*}


\subsection{Пример | И. А. Лавров, Л. Л. Максимова: номер 13(е)}
Доказать, что
\[
    (A \cap B) \cup C = A \cap (B \cup C) \Leftrightarrow C \subseteq A
\]

\subsubsection{Доказательство 1 $\Leftarrow$ 2}
\paragraph{Дано}
\[
C \subseteq A
\]

\paragraph{Доказать}
\[
    (A \cap B) \cup C = A \cap (B \cup C)
\]

\paragraph{Доказательство}
\begin{gather*}
    C \subseteq A \Rightarrow C \cup A = A, C \cap A = C. \\
    (A \cap B) \cup C = (A \cup C) \cap (B \cup C) = A \cap (B \cup C)
\end{gather*}


\subsubsection{Доказательство 1 $\Rightarrow$ 2}
\paragraph{Дано}
\[
    (A \cap B) \cup C = A \cap (B \cup C)
\]

\paragraph{Доказать}
\[
    C \subseteq A
\]

\paragraph{Доказательство}
\begin{gather*}

\section{Основные свойства}

\subsection{Определение | Рефлексивность}
Бинарное отношение $\rho$ на множестве $X$ называется рефлексивным, если $\forall x \in X: x \rho x$.

\subsubsection{Пример}
\begin{gather*}
    \le, \mathbb{R}. ~ \forall x \in \mathbb{R}: x \le x \\
    x \rho y \Leftrightarrow x + y = 1, \mathbb{R} \\
    \langle 0, 0 \rangle \notin \rho, 0+0 \neq 1
\end{gather*}


\subsection{Определение | Симметричность}
Бинарное отношение $\rho$ на множестве $X$ называется симметричным, если $\forall x, y \in X: x \rho y \Rightarrow y \rho x$.

\subsubsection{Пример}
\begin{gather*}
    \le, \mathbb{R}. ~ \forall x, y \in X: \\
    x \rho y \Leftrightarrow x + y = 1, \mathbb{R} \\
    \le, \mathbb{R}. ~ x + y = 1 \Rightarrow y + x = 1 \\
    1 \le 2 \nRightarrow 2 \le 1
\end{gather*}


\subsection{Определение | Антисимметричность}
Бинарное отношение $\rho$ на множестве $X$ называется антисимметричным, если 
\[
    \forall x, y \in X: x \rho y, y \rho x \Rightarrow x = y.
\]

\subsubsection{Пример}
\begin{gather*}
    \le, \mathbb{R}. ~ x \le y, y \le x \Leftrightarrow x = y \\
    x \rho y \Leftrightarrow x + y = 1, \mathbb{R} \\
    \langle 0, 1 \rangle \in \rho, \langle 1, 0 \rangle \in \rho, 1 \neq 0
\end{gather*}


\subsection{Определение | Асимметричность}
Бинарное отношение $\rho$ на множестве $X$ называется асимметричным, если
\[
    \forall x, y \in X: x \rho y \Rightarrow y \overline{\rho} x.
\]

\subsubsection{Пример}
\begin{gather*}
    <, \mathbb{R}. ~ x < y ~\Rightarrow y \cancel{<} x \\
    x \rho y \Leftrightarrow x + y = 1, \mathbb{R} \\
    \langle 0, 1 \rangle, \langle 1, 0 \rangle \in \rho
\end{gather*}


\subsection{Определение | Транзитивность}
Бинарное отношение $\rho$ на множестве $X$ называется транзитивным, если
\[
\forall x, y, z \in X: x \rho y, y \rho z \Rightarrow x \rho z.
\]

\subsubsection{Пример}
\begin{gather*}
    <, \mathbb{R}. ~ x < y, y < z \Rightarrow x < z \\
    \text{любовь, люди} \\
    \text{Петя любит Машу, Маша любит Васю } \cancel{\Rightarrow} \text{ Петя любит Васю}
\end{gather*}

\]

    \forall x \in \mathbb{R}: x = x \\
    \forall x, y \in \mathbb{R}: x = y, y = x \Rightarrow x = y \\
    \forall x, y, z \in \mathbb{R}: x = y, y = z \Rightarrow x = z
\end{gather*}

\[

\]
\begin{gather*}
    \forall x \in \mathbb{R}: x \le x \\

    \forall x, y, z \in \mathbb{R}: x \le y, y \le z \Rightarrow x \le z
\end{gather*}

\[

\]

\subsection{Определение}
Класс функций называется функционально-замкнутым (ФЗК), если любые суперпозиции функций этого класса остаются в этом классе.

\subsubsection{Примеры}
\begin{enumerate}
    \item $K$ --- все булевые функции --- ФЗК
    \item $P_1$ --- функции одной переменной --- ФЗК \\
    $f(x), g(x) \in P_1; f(g(x)) \in P_1$
    \item $P_2$ --- функции двух переменных --- не ФЗК \\
    $f_1(x, y), f_2(x, y) \in P_2; f_1(x, f_2(y, z)) \notin P_2$
    \item $T_{\overline{1}}$ --- $f(1, 1, \dots, 1) = 0$ \\
    $+ \in T_{\overline{1}}$ ~~~ $1 + (1 + 1) = 1 \notin T_{\overline{1}}$
\end{enumerate}


\subsection{5 основных ФЗК}
\begin{enumerate}
    \item $T_0$ --- сохраняет $0: f(0, 0, \dots, 0) = 0$ \\
    $\& \in T_0: 0 \& 0 = 0$ \\
    $\subset \notin T_0: 0 \subset 0 = 1$
    \item $T_1$ --- сохраняет $1: f(1, 1, \dots, 1) = 1$ \\
    $\& \in T_1: 1 \& 1 = 1$ \\
    $\subset \notin T_1: 1 \subset 1 = 1$
    \item $S$ --- самодвойственная \\
    $f^*(x_1, \dots, x_n) = \neg f(\neg x_1, \dots, \neg x_n)$ \\
    $f$ --- самодвойственна, если $f = f^*$ \\
    $(x \& y)^* = \neg(\neg x \& \neg y) = x \vee y                          ~~~~~ \&,   \vee \notin S$ \\
    $(x \sim y)^* = \neg(\neg x \sim \neg y) = x + 1 + y + 1 + 1 + 1 = x + y ~~~~~ \sim, +    \notin S$ \\
    $(1)^* = \neg 1 = 0                                                      ~~~~~ 1,    0    \notin S$ \\
    $(\neg x)^* = \neg(\neg(\neg x)) = \neg x                                ~~~~~ \neg       \in    S$ \\
    \item $L$ --- линейные функции \\
    $f \in L$, если $f$ --- линейный множитель Жегалкина. \\
    $f(x_1, \dots, x_n) = a_0 + a_1 x + \cdot + a_n x_n$ \\
    $\vee \notin L: x \vee y = xy + x + y \notin L$ \\
    $\sim \in    L: x \sim y = x + y + 1  \in    L$ \\
    \item $M$ --- монотонные функции \\
    $\langle t \rangle = \langle t_1, \dots, t_n \rangle; \langle s \rangle = \langle s_1, \dots, s_n \rangle$ \\
    $\text{Частичный порядок: } \langle t \rangle \preccurlyeq \langle s \rangle \Leftrightarrow t_i \le s_i, \forall i \in \overline{1, n}$ \\
    $\langle 0, 1, 0 \rangle \preccurlyeq \langle 1, 1, 0 \rangle: \langle 0, 1, 0 \rangle \text{ и } \langle 1, 0, 0 \rangle \text{ --- не сравнимы}$ \\
    $ f(x_1, \dots, x_n) \in M, \text{если } \forall \langle t \rangle \preccurlyeq \langle s \rangle \Rightarrow f(t) \le f(s)$
\end{enumerate}
