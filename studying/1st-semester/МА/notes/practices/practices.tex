\subsection{Сложение}
\begin{gather*}

\end{gather*}

\subsubsection{Пример}
\[

\subsubsection{Пример}

\subsection{Умножение}
\[
    c_1 c_2 = (x_1 + iy_1)(x_2 + iy_2) = 
    x_1 x_2 + x_1 i y_2 + x_2 i y_1 + i^2 y_1 y_2 = 
    \underbrace{(x_1 x_2 = y_1 y_2)}_{Re} + \underbrace{i(x_1 y_2 + x_2 y_1)}_{Im}
\]

\subsubsection{Пример}
\[
    c_{11} c_{22} = (5 - 8i)(3 + i) = 15 + 5i - 24i - 8i^2 = 15 - 19i + 8 = 25 - 19i
\]

\subsubsection{Пример}

\[

\]

\subsubsection{Пример}
\[
    c_{11} c^*_{11} = (5 - 8i)(5 + 8i) = 25 - 40i + 40i - 64i^2 = 25 + 64 = 89

\subsection{Деление}
\[
    \frac{c_1}{c_2} = 
    \frac{c_1}{c_2} \cdot \frac{c^*_2}{c^*_2} = 
    \frac{x_1 + iy_1}{x_2 + iy_2} \cdot \frac{x_2 - iy_2}{x_2 - iy_2} =
    \frac{x_1 x_2 - i x_1 y_2 + i x_2 y_1 - i y_1 y_2}{x^2_2 + y^2_2} =
    \frac{x_1 x_2 + y_1 y_2}{x^2_2 + y^2_2} + i \cdot \frac{x_2 y_1 - x_1 y_2}{x^2_2 + y^2_2}

\section{Тригонометрическая форма записи комплексного числа}
\[
    c = r(\cos \varphi + i \cdot \sin \varphi)
\]

Характеризуется радиус-вектором и углом
\[
    |c| = r = \sqrt{x^2 + y^2}
\]

$\varphi$ --- аргумент $c$

\begin{gather*}
    \tg \varphi = \frac{y}{x} ~~~ \varphi = \arctg (\frac{y}{x}) \\
    x = r \cdot \cos \varphi ~~~ y = r \cdot \sin \varphi \\
    c = r \cdot \cos \varphi + i \cdot r \cdot \sin \varphi \\
    c = r(\cos \varphi + i \cdot \sin \varphi)
\end{gather*}


\subsection{Формула Муавра}
