\[
    A \times B = \{(x, y): x \in A \wedge x \in B\}
\]

    A \times B = \{(x, y): x \in [1, 2] \wedge y \in [2, 3]\}
\end{gather*}

\section{Существование точных граней множества}

\subsection{Определение}
Наименьшее из мажорант множества называется точной верхней гранью этого множества.


\subsection{Утверждение}
Любое ограниченное сверху множество имеет точную вернюю грань --- супремум.

\subsubsection{Доказательство}
\begin{gather*}
    \text{Пусть } A \text { --- ограниченное множество, } B \text { --- множество его мажорант, тогда} \\ 
    \forall x \in A \text{ и } \forall y \in B: x = y. \\
    \text{По аксиоме 16 } \exists c \in \mathbb{R}: \forall x \in A \text{ и } \forall y \in B: x \le c \le y, \\
    \text{ то есть } c \text{ --- мажоранта } A, \text{ но любая другая мажоранта больше } c, \text{ то есть } c = \sup A.
\end{gather*}


\subsection{Определение}
Наибольший из минорант множества называется точной нижней гранью этого множества - инфинум $c = \inf A$.


\subsection{Утверждение}
Любое ограниченное снизу множество имеет точную нижнюю грань.



\section{Принцип вложенных отрезков}

\subsection{Определение}
Последовательностью называется отображение множества натуральных чисел в множество любой природы.

\subsubsection{Пример} 
\dots

Пусть $\{I_n\}: \forall n \in \mathbb{N} ~ I_{n+1} < I_n$, тогда $\{I_n\}$ называется системой вложенных отрезков.


\subsection{Теорема | Проекция вложенных отрезков (принцип Коша Кантора)}
Пусть $\{U_n\}, n \in \mathbb{N}$ --- система вложенных отрезков, тогда $\exists c \in \mathbb{R} ~ \forall n \in \mathbb{N}: c \in U_n$. \\
Кроме того, если $U_n = [a_n ; b_n]$ и $\forall \epsilon > 0 ~ \exists n \in \mathbb{N}$, что $b_n - a_n < \epsilon$, то эта точка --- единственная.

\subsubsection{Доказательство}
\begin{gather*}
    \text{Рассмотрим множества } \{a_1, a_2, ..., a_n, ...\} \text{ и } \{b_1, b_2, ..., b_n, ...\} \\
    \text{Очевидно, что } \forall k, m: a_k < b_m \\
    \exists c ~ \forall k, m \in \mathbb{N}: a_k \le c \le b_m \Rightarrow \forall n \in \mathbb{N}: c \in U_n \\
    \text{Пусть } \forall \epsilon > 0 ~ \exists n \in \mathbb{N}: b_n - a_n < \epsilon \\
    \text{Пусть } c_1 \neq c, c_1 \in U_n; n \in \mathbb{N} \\
    \text{Обозначим } \epsilon^{-1} = |c_1 - c| \\
    \exists n \in \mathbb{N}, \text{что } b_n - a_n < \epsilon \text{, то есть } c_1 \text{ и } c \text{ не могут одновременно принадлежать отрезку } U_n
\end{gather*}

\subsubsection{Пример}
\[
U_n = \left[\frac{1}{2} - \frac{1}{n} ; \frac{1}{2} + \frac{1}{n}\right], c = \frac{1}{2}
\]


\subsection{Определение}
Система множеств $A_1$, $A_2$, ..., $A_n$, ... называестя покрытием множества $B$, если $\forall x \in B ~ \exists n: x \in A_n$.


\subsection{Определение}
Подпокрытием называется набор множеств покрытия, также являющийся покрытием.


\subsection{Пример}
\begin{gather*}
    B = \left[0, 1\right],~A_1 = \left(-\frac{1}{2} ; \frac{2}{3}\right),~A_2 = \left(\frac{1}{3} ; \frac{4}{3}\right),~A_3 = \left(\frac{1}{2} ; \frac{2}{3}\right) \\
    A_1, A_2 \text{ --- подпокрытие} \\
    A_1, A_2, A_3 \text{ --- покрытие}
\end{gather*}


\subsection{Теорема | Принцип конечного подпокрытия (принцип Бареля Лебега)}
Из любого покрытия отрезка интервалами можно выделить конечные покрытия.

\subsubsection{Пример}
\includegraphics[scale=0.4]{images/2024-09-27/01.jpg}

\subsubsection{Доказательство}
Пусть существует бесконечное покрытие отрезка, из которого нельзя выделить конечное покрытие отрезка.

Разделим пополам исходный отрезок.

Тогда для хотя бы одной половины тоже требуется бесконечность покрытия.

Разделим и его пополам и убедимся в том, что одна из новых половин также требует бесконечное покрытие.

Продолжим процесс деления пополам до бесконечности.

Заметим, что выделенная каждый раз половинка образует вложенную систему отрезков, пересечением которых будет некоторая единственная точка $C$, принадлежащая какому-то интервалу из исходного покрытия.

Так как длина половинки может быть как угодно малой, какая-то из половинок целиком попадёт внутрь этого интервала, то есть этот интервал уже является покрытием соответствующей половинки, а вовсе не бесконечного множества интервалов.


\subsection{Определение}
Окрестностью точки называется любой интервал, содержащий эту точку


\subsection{Определение}
Проколотой окрестностью точки называется окрестность точки, из которой исключена сама точка


\subsection{Определение}
Граничной точкой множества называется такая его точка, любая окрестность которой содержит как элементы множества, так и элементы, не принадлежащие множеству


\subsection{Определение}
Внутренней точкой множества называется точка множества, принадлежащая ему вместе с некоторой своей окрестностью.


\subsection{Определение}
Предельной точкой множества называется такая его точка, в любой окрестности которой содержится бесконечно много точек множества.

\subsubsection{Пример}
\includegraphics[scale=0.2]{images/2024-09-27/02.jpg}


\subsection{Определение}
Изолированной точкой множества называется такая точка этого множества, у которой существует проколотая окрестность, не содержащая элементов множеств.


\subsection{Теорема | Принцип существования}
Любое бесконечное ограниченное множество имеет предельную точку

\subsubsection{Доказательство}
$A$ --- бесконечное множество, $\exists [a, b]: A \subset [a, b]$.

Пусть $U_n$ --- произвольное покрытие отрезка $[a, b]$ интервалами, тогда $U_n$ --- покрытие множества $A$.

Пусть $A$ не имеет предельных точек, тогда любой интервал из $U_n$ содержит не более чем конечное множество точек из множества $A$.

Из $U_n$ можно выделить конечное подпокрытие, то есть $A$ --- конечное множество.

\part{Предел последовательности}

\section{Введение}

\subsection{Определение}
Числовой последовательностью называется отображение множества $\mathbb{N}$ в множестве $\mathbb{R}$.
\[
\{a_n\}, a_n \in \mathbb{R}, n \in \mathbb{N}
\]

\subsubsection{Пример}
\begin{gather*}
    a_n = 1 \\
    a_n = (-1)^n \\
    a_n = \frac{1}{n} \\
    a_n = 1 + (-1)^n
\end{gather*}


\subsection{Определение}
Последовательность называется ограниченной, если множество её значений ограничено.


\subsection{Определение}
Число $A$ называется пределом последовательности $a_n$, если какова бы не была окрестность точки $A$, найдётся такое $N$, что $\forall n \in \mathbb{N}, n > N, a_n$ находится в этой окрестности.
\[
\lim_{n \to \infty} a_n = A
\]


\subsection{Определение}
\[
(\lim_{n \to \infty} a_n = A) \Leftrightarrow (\forall \epsilon > 0 ~ \exists N(\epsilon), \forall n \in \mathbb{N}: |a_n - A| < \epsilon)
\]


\subsection{Утверждение}
Последовательность может иметь не более одного предела

\subsubsection{Доказательство}
\begin{gather*}
    \text{Пусть } \exists \lim_{n \to \infty} a_n = A_1, \exists \lim_{n \to \infty} a_n = A_2, A_1 < A_2 \\
    \text{Пусть } \epsilon = \frac{A_2 - A_1}{3} \\
    \epsilon N_1 = \forall n > N_1: |a_1 - A_1| < \epsilon \Leftrightarrow A_1 - \epsilon < a_n < A_1 + \epsilon \\
    \epsilon N_2 = \forall n > N_2: |a_2 - A_2| < \epsilon \Leftrightarrow A_2 - \epsilon < a_n < A_2 + \epsilon \\
    \text{Пусть } N = \max\{N_1, N_2\}: \\
    \forall n > N: a_1 < A_1 + \epsilon = A_1 + \frac{A_2 - A_1}{3} = \frac{2A_1 + A_2}{3} < \frac{A_1 + 2A_2}{3} = A_2 - \epsilon < a_n    
\end{gather*}


\subsubsection{Замечание}
Ограниченная последовательность может не иметь предел

\subsubsection{Пример}
\includegraphics[scale=0.2]{images/2024-10-04/01.jpg}


\subsection{Определение}
\begin{gather*}
    (\lim_{n \to \infty} a_n = \infty) \Leftrightarrow (\forall \epsilon > 0 ~ \exists N(\epsilon) ~ \forall n > N: |a_n| > \epsilon) \\
    (\lim_{n \to \infty} a_n = +\infty) \Leftrightarrow (\forall \epsilon > 0 ~ \exists N(\epsilon) ~ \forall n > N: a_n > \epsilon) \\
    (\lim_{n \to \infty} a_n = -\infty) \Leftrightarrow (\forall \epsilon > 0 ~ \exists N(\epsilon) ~ \forall n > N: a_n > -\epsilon) \\
    (\{a_n\} \text{ является сходящей}) \Leftrightarrow (\forall \epsilon > 0 ~ \exists N(\epsilon), m \in \mathbb{N}, n > m > N: |a_n - a_m| < \epsilon)
\end{gather*}


\subsection{Определение}
Если предел последовательности равен $\infty$, $+\infty$ или $-\infty$ последовательность называется \\
бесконечно большой.


\subsection{Определение}
Последовательность называется сходящейся, если она имеет конечный (числовой) предел.


\subsection{Определение}
Если последовательность имеет предел равный нулю, то она называется бесконечно малой \\
последовательностью.


\subsection{Определение}
Последовательность $\{a_n\}$ называется фундаментальной, если 
\[
\forall\epsilon > 0, \exists{N(\epsilon)}: \forall{m} > n > N: |a_m - a_n| < \epsilon
\]


\subsection{Критерий Коши} 
(последовательность является фундаментальной) $\Leftrightarrow$ (последовательность является сходящейся)

\subsubsection{Доказательство $1 \Rightarrow 2$}
\begin{gather*}
    \exists \lim_{n\to\infty} a_n = A, A \in \mathbb{R} \\
    \forall\epsilon > 0~\exists{N(\epsilon)} \\
    \forall{n} > N, n \in \mathbb{N}: |a_n - A| < \frac{\epsilon}{2} \\
    \forall{m} > N, n \in \mathbb{N}: |a_m - A| < \frac{\epsilon}{2} \\
    |a_n - a_m| = |a_n - A - (a_m - A)| \le |a_n - A| + |a_m - A| < \frac{\epsilon}{2} + \frac{\epsilon}{2} = \epsilon \\
\end{gather*}

\subsubsection{Доказательство $1 \Leftarrow 2$}
\begin{gather*}
	\forall \epsilon > 0~\exists N(\epsilon): \forall m > n > N : |a_n - a_m| < \frac{\epsilon}{3} \\
	\text{Пусть } x_n =\inf_{m \ge n} a_m \\
	\text{Пусть } y_n =\sup_{m \ge n} a_m \\
	x_n \le x_{n+1} \le y_{n+1} \le y_n \\
	\{[x_n, y_n]\} \text{ --- система вложенных отрезков} \Rightarrow \exists c: \forall n \in \mathbb{N}, c \in [x_n, y_n] \\
	m > n \\
	a_n - \frac{\epsilon}{3} < a_m < a_n + \frac{\epsilon}{3} \\
	a_n - \frac{\epsilon}{3} \le x_m \le c \le y_m \le a_n + \frac{\epsilon}{3} \\
	|a_n - c| \le \frac{\epsilon}{3} \\
	|a_m - a_n| < \frac{\epsilon}{3} \\
	|x_m - a_n| \le \frac{\epsilon}{3} \\
	|y_m - a_n| \le \frac{\epsilon}{3} \\
	|a_m - c| \le |a_m - a_n + a_n - c| \le |a_m - a_n| + |a_n - c| \le
	\frac{\epsilon}{3} + \frac{\epsilon}{3} = \frac{2\epsilon}{3} < \epsilon
\end{gather*}

\includegraphics[scale=0.2]{images/2024-10-04/02.jpg}

\subsubsection{Следствие}
Область значений последовательности есть ограниченное множество, то есть последовательность ограничена.



\section{Алгебраические свойства пределов последовательности}
Пусть $\lim_{n \to \infty} a_n = A, \lim_{n \to \infty} b_n = B$

\subsection{Утверждение | Сложение}
$\exists \lim_{n \to \infty} (a_n + b_n) = A + B$

\subsubsection{Доказательство}
\begin{gather*}
	\text{Если } (\lim_{n \to \infty} a_n = A) \Leftrightarrow (\forall \epsilon > 0 \text{ } \exists N_1(\epsilon): \forall n > N_1, n \in \mathbb{N} : |a_n - A| < \frac{\epsilon}{2}) \\
	(\lim_{n \to \infty} b_n = B) \Leftrightarrow (\forall \epsilon > 0 \text{ } \exists N_2(\epsilon): \forall n > N_2, n \in \mathbb{N} : |b_n - B| < \frac{\epsilon}{2}) \\
	\text{Пусть } N = \max\{N_1, N_2\} \\
	\forall n > N : |a_n + b_n - (A + B)| \le |a_n - A| + |b_n - B| \le \frac{\epsilon}{2} + \frac{\epsilon}{2} = \epsilon \Leftrightarrow \lim_{n \to \infty} (a_n + b_n) = A + B
\end{gather*}


\subsection{Утверждение | Умножение}
$\exists \lim_{n \to \infty} (a_n \cdot b_n) = A \cdot B$

\subsubsection{Доказательство}
\begin{gather*}
	a_n \cdot b_n = (a_n - A + A)(b_n - B + B) = (a_n - A)(b_n - B) + A(b_n - B) + B(a_n - A) + AB \\
	\forall \epsilon > 0 ~ \exists N_1(\epsilon): \forall n > N_1 : |a_n - A| < \epsilon \\
	\forall \epsilon > 0 ~ \exists N_2(\epsilon): \forall n > N_2 : |b_n - B| < \epsilon \\
	\forall \epsilon > 0 ~ \exists N_3(\epsilon): \forall n > N_3 : |b_n - B| < \frac{\epsilon}{3|A|} \text{ (если $A \neq 0$, иначе отсутствует)} \\
	\forall \epsilon > 0 ~ \exists N_4(\epsilon): \forall n > N_4 : |a_n - A| < \frac{\epsilon}{3|B|} \text{ (если $A \neq 0$, иначе отсутствует)} \\
	\text{let } N = \max\{N_1, N_2, N_3, N_4\} \\
	\forall n > N: |a_n b_n - AB| \le |a_n - A||b_n - B| + |A||b_n - B| + |B||a_n - A| < \\
	< \epsilon^2 + |A| \frac{\epsilon}{3|A|} + |B| \frac{\epsilon}{3|B|} < 3 \cdot \frac{\epsilon}{3} = \epsilon \\
	(\epsilon^2 < \frac{\epsilon}{3} \Leftarrow \epsilon < \frac{1}{3})
\end{gather*}

\subsection{Утверждение | Деление}
Если $B \neq 0$, то $\exists \lim_{n \to \infty} \frac{a_n}{b_n} = \frac{A}{B}$

\subsubsection{Доказательство}
\begin{gather*}
\frac{a_n}{b_n} = \frac{A}{B} = \frac{a_n \cdot B - b_n \cdot A}{b_n \cdot B} = \frac{(a_n - A) \cdot B + A \cdot B - (b_n - B) \cdot A - A \cdot B}{b_n \cdot B} = \frac{a_n - A}{b_n} - \frac{b_n - B}{b_n} \cdot \frac{A}{B} \\
\exists \epsilon ~ N_1 \forall n > N_1 |b_n| > \frac{B}{2} \\
\exists \epsilon ~ N_2 \forall n > N_2 |a_n - A| < \frac{\epsilon|A|}{4} \\
\exists \epsilon ~ N_3 \forall n > N_3 |b_n - B| < \frac{\epsilon|B|^2}{4|A|} \\
\text{Пусть } N = max\{N_1, N_2, N_3\}, \forall n > N: \\
\frac{|a_n - A|}{b_n - B} \le {} \frac{|a_n - A|}{|b_n|} + \frac{|b_n - B|}{b_n} \cdot \frac{|A|}{|B|} \le \frac{2 \cdot \epsilon \cdot |B|}{4|B|} + \frac{2 \cdot \epsilon* |B|^2}{4|B|} \cdot \frac{|A|}{|B|} = \frac{\epsilon}{2} + \frac{\epsilon}{2} = \epsilon
\end{gather*}



\section{Предельный переход в неравенства}

\subsection{Утверждение}
\begin{gather*}
    \text{1. } \exists \lim_{n \to \infty} a_n = A \\
    \text{2. } \exists \lim_{n \to \infty} b_n = B \\
    \text{3. } \forall n \in \mathbb{N}: a_n \le c_n \le b_n
\end{gather*}

\subsubsection{Доказательство $\Rightarrow$}
$\exists \lim_{n \to \infty} c_n = A$



\subsection{Утверждение}
\begin{gather*}
    \text{1. } \exists \lim_{n \to \infty} a_n = A \\
    \text{2. } \exists \lim_{n \to \infty} b_n = B \\
    \text{3. } \forall n \in \mathbb{N}: a_n < b_n
\end{gather*}

\subsubsection{Доказательство $\Rightarrow$}
\begin{gather*}
    \forall \epsilon > 0 ~ \exists N_1(\epsilon): \\
    \forall n > N_1: |a_n - A| < \epsilon \Leftrightarrow A - \epsilon < a_n < A + \epsilon \\
    \forall n > N_2: |b_n - B| < \epsilon \Leftrightarrow B - \epsilon < b_n < B + \epsilon \\
    c_n = b_n - a_n, \exists \lim_{n \to \infty} c_n = B - a, c_n \ge 0, \forall n \in \mathbb{N} \\
    \text{Пусть } B - A < 0: \epsilon = \frac{1}{2} \cdot |B - A| \\
    \exists N ~ \forall n > N: c_n < B - A + \epsilon = B - A + \frac{1}{2} \cdot (A - B) = \frac{1}{2} \cdot B - \frac{1}{2} \cdot A < 0 \\
    b_n - a_n < 0 \text{ --- противоречие}
\end{gather*}


\subsection{Утверждение | Принцип двух полицейских}
\begin{gather*}
	\begin{cases}
		\exists \lim_{n \to \infty} a_n = A \\
		\exists \lim_{n \to \infty} b_n = A \\
		\forall n \in \mathbb{N}: a_n \le c_n \le b_n
	\end{cases} \Rightarrow \lim_{c \to \infty} c_n = A
\end{gather*}

\subsubsection{Доказательство}
\begin{gather*}
    \forall \epsilon ~ \exists N(\epsilon) ~ \forall n > N_1 \\
    \forall \epsilon ~ \exists N(\epsilon) ~ \forall n > N_2 \\
    N = \max{N_1, N_2} \\
    \forall n > N: A - \epsilon < a_n \le c_n \le b_n < A + \epsilon \\
    A - \epsilon < c_n < A + \epsilon \\
    |c_n - A| < \epsilon
\end{gather*}



\section{Монотонные последовательности}

\subsection{Определение}
Последовательность $\{a_n\}$ называется возрастающей, если для $\forall n \in \mathbb{N} ~ a_n < a_n+1$.


\subsection{Определение}
Последовательность $\{a_n\}$ называется неубывающей, если для $\forall n \in \mathbb{N} ~ a_n \le a_n+1$.


\subsection{Определение}
Последовательность $\{a_n\}$ называется убывающей, если для $\forall n \in \mathbb{N} ~ a_n > a_n+1$.


\subsection{Определение}
Последовательность $\{a_n\}$ называется невозрастающей, если для $\forall n \in \mathbb{N} ~ a_n \ge a_n+1$.


\subsection{Теорема | Вейерштрасс}
Монотонная ограниченная последовательность имеет конечный предел.

\subsubsection{Доказательство}
\begin{gather*}
    \text{Пусть для последовательности } \{a_n\}: \\
    \forall{n} \in \mathbb{N}: a_{n+1} \ge a_n, \\
    \forall{n} \in \mathbb{N}: a_n < A, \text{тогда} \\
    \text{1. } a_n \text{ --- финальная постоянная последовательность, то есть } \exists{N} ~ \forall{n} > N: a_n = a \in \mathbb{R} \\
    \lim_{n\to\infty} a_n \\
    \text{2. Множество значений } a_n \text{ бесконечно, то есть множество значений имеет предельную точку,} \\ 
    \text{так как $a_n$ --- ограничена} \\
    \text{3. } \exists c = \sup_{n \in \mathbb{N}} a_n ~ \forall \epsilon > 0 ~ \exists a_n: c - \epsilon < a_n \le c, \forall m > n: \\
    c - \epsilon < a_n \le a_m \le c \rightarrow \lim_{n \to \infty} a_n = c
\end{gather*}


\subsection{Утверждение}
Неубывающая последовательность, имеющая предел, ограничена сверху.


\subsection{Утверждение}
Для того, чтобы монотонная последовательность имела предел, необходимо и достаточно, чтобы она была ограничена.

\subsubsection{Пример}
\begin{gather*}
a_n = \frac{n}{q^n}, q > 1 \\

\subsubsection{Доказательство}
Поделим отрезок $[a, b]$ пополам:
\begin{enumerate}
    \item $f(\frac{a + b}{2}) = 0$, то $c = \frac{a + b}{2}$
    \item $f(\frac{a + b}{2}) \neq 0$, то значение функции $f(x)$ на концах одной из половинок имеют разные знаки. \\
    Обозначим эту половинку как $[a_1, b_1]$. \\
    Для отрезка $[a_1; b_1]$ повторим процедуру. \\ 
    В результате либо найдм точку с, либо найдём отрезок $[a_2, b_2]$, обладающий тем же свойством. \\
    Будем повторять эту процедуру либо до нахождения точки $c$, либо либо до бесконечности. \\
    Предположим, пришлось выполнять процедуру до бесконечности. \\
    В результате получаем последовательность вложенных отрезков. \\
\end{enumerate}
\begin{gather*}
    [a_2, b_2], [a_3, b_3], \dots, [a_n, b_n], \dots \\
    \lim_{n \to \infty} (b_n - a_n) = 0 \\

    \exists \lim_{n \to \infty} a_n = c ~~~ f(a_n) < 0 \\
    \exists \lim_{n \to \infty} b_n = c ~~~ f(b_n) > 0 \\
    f(x) \in C(c) ~~~~ \exists \lim_{x \to c} f(x) = f(c) \\
    \lim_{n \to \infty} f(a_n) = f(c) \le 0, \lim_{n \to \infty} f(b_n) = f(c) \ge 0
\end{gather*}

\section{Равномерная непрерывность}
Пусть $f: E \rightarrow \mathbb{R}$

\subsection{Определение}
Функция $f(x)$ называется равномерно непрерывной на $E$, если
\[
    \forall \epsilon > 0 ~ \exists \delta(\epsilon) > 0 ~ \forall x_1, x_2 \in E, |x_1 - x_2| < \delta \Rightarrow |f(x_1) - f(x_2)| < \epsilon
\]


\subsection{Утверждение | Связь определения с обычной непрерывностью}
Если $f$ равномерно непрерывна на $E$, то $f$ непрерывна в любой точке $E$.

\subsubsection{Доказательство}
\begin{gather*}
    \text{Пусть } x_o \in E ~ \forall \epsilon > 0 ~ \exists \delta(\epsilon) > 0 ~ \forall x \in E,
    |x - x_0| < \delta: |f(x) - f(x_0)| < \epsilon, \\
    \text{ то есть } \exists \lim_{x \to x_0} f(x) = f(x_0), f(x) \in C(x_0) 
\end{gather*}
Из непрерывности не следует равномерная непрерывность.

\subsubsection{Пример}
\begin{gather*}
    y = x^2, x \in \mathbb{R} \\
    y = x^2 \in C(\mathbb{R}) \\
    x_1 = \sqrt{n}, ~ x_2 = \sqrt{n + 1} \\
    f(x_2) - f(x_1) = 1 \\
    x_2 - x_1 = \sqrt{n + 1} - \sqrt{n} \\
    \lim_{n \to \infty} (\sqrt{n + 1} - \sqrt{n}) = 
    \lim_{x \to \infty} \frac{1}{\sqrt{n + 1} + \sqrt{n}} \\
    y = x^2 \text{ не является равномерно непрерывной на } \mathbb{R}
\end{gather*}

\subsubsection{Пример}
\begin{gather*}
    y = \frac{1}{x}, x \in (0; 1) \\
    x_1 = \frac{1}{n + 1} ~~ y_1 = n + 1 \\
    x_2 = \frac{1}{n}     ~~ y_2 = n \\
    y_1 - y_2 = 1 \\
    \lim_{n \to \infty} (x_2 - x_1) = 
    \lim_{n \to \infty} (\frac{1}{n} - \frac{1}{n + 1}) =
    \lim_{n \to \infty} \frac{1}{n(n + 1)} = 0 \\
    y = \frac{1}{x} \text{ не является равномерно непрерывной на } (0; 1)
\end{gather*}



\subsection{Теорема равномерной непрерывности}
Если функция непрерывна на отрезка, то она равномерно непрерывна на этом отрезке.
