\subsection{Теория}
\paragraph{$\boxed{1}$}
\begin{gather*}
    \sqrt{a^2 - x^2} = \sqrt{a^2 - a^2 \sin^2 t} = a \sqrt{1 - \sin^2 t} = a \cos t \\

\end{gather*}
\paragraph{$\boxed{2}$}
\begin{gather*}
    \sqrt{x^2 - a^2} = \sqrt{\frac{a^2}{\cos^2 t} - a^2} = a \sqrt{\frac{1 - \cos^2 t}{\cos^2 t}} = a \tg t \\
    x = a \sec t = \frac{a}{\cos t} ~~~~~ dx = \frac{a \sin t}{\cos^2 t} dt
\end{gather*}
\paragraph{$\boxed{3}$}
\begin{gather*}
    \sqrt{x^2 + a^2} = \sqrt{\frac{a^2 \sin^2 t}{\cos^2 t} + a^2} =
    a \sqrt{\frac{\sin^2 t + \cos^2 t}{\cos^2 t}} = \frac{a}{\cos t} \\
    x = a \tg t ~~~~~ dx = \frac{a}{\cos^2 t} dt
\end{gather*}


\subsection{Практика}

\subsubsection{Пример}
\[
    \int \frac{x^2 dx}{\sqrt{1 - x^2}}
\]
\begin{gather*}
    \text{Замена: } x = \sin x ~~~~~ dx = \cos t dt \\
      \int \frac{\sin^2 t \cos t dt}{\cos t} =
      \int \sin^2 t dt =
      \int \frac{1 - \cos 2t}{2} dt =                           \\
    = \frac{t}{2} - \int \frac{\cos 2t}{2} dt =
      \frac{t}{2} + \frac{1}{2} \cdot \frac{1}{2} \sin 2t + C =
      \frac{1}{2} \arcsin x + \frac{1}{2} x \sqrt{1 - x^2} + C  \\  
\end{gather*}

\subsubsection{Пример}
\[
    \int \frac{\sqrt{x^2 + 1}}{x} dx
\]
\begin{gather*}
    \text{Замена: } x = \tg t ~~~~~ dx = \frac{dt}{\cos^2 t} \\
      \int \frac{1}{\cos t \tg t \cos^2 t} dt =
      \int \frac{dt}{\sin t \cos^2 t} =
      \int \frac{\sin^2 t + \cos^2 t}{\sin t \cos^2 t} dt =
      \int \frac{\sin t}{\cos^2 t} dt + \int \frac{1}{\sin t} dt = \\
    = -\int \frac{1}{\cos^2 t} d(\cos t) + \int \frac{1}{\sin t} dt =
      \frac{1}{\cos t} + \ln \left|\tg \frac{t}{2}\right| + C = 
      \sqrt{x^2 + 1} + \ln \left|\sqrt{x^2 + 1} \cdot \frac{1}{x} - \frac{1}{x}\right| + C
\end{gather*}

\subsubsection{Пример}
\[
    \int \frac{\sqrt{x^2 - 4}}x dx
\]
\begin{gather*}
    \text{Замена: } x = \frac{2}{\cos t} ~~~~~ dx = \frac{2 \sin t}{\cos^2 t} dt \\
      \int \frac{\sqrt{\frac{4}{\cos^2 t} - 4}}{\frac{2}{\cos t}} \cdot \frac{2 \sin t}{\cos^2 t} dt =
      2 \int \sqrt{\frac{1}{\cos^2 t} - 1} \cdot \frac{\sin t}{\cos t} dx =
      2 \int \sqrt{\frac{1 - \cos^2 t}{\cos^2 t}} \cdot \frac{\sin t}{\cos t} dt =
      2 \int \frac{\sin^2 t}{\cos^2 t} dt = \\
    = 2 \int \frac{1 - \cos^2 t}{\cos^2 t} dt =
      2 \int \left(\frac{1}{\cos^2 t} - 1\right) dt =
      2 \tg t - 2t + C = \sqrt{x^2 - 4} - 2 \arccos \frac{2}{x} + C
\end{gather*}

    = \frac{3}{2} \ln \left|x^2 - 4x + 5\right| + 4 \arctg (x - 2) + C
\end{gather*}

\[

\]
